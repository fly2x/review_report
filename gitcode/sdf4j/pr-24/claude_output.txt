```
        this.l = l;
    }
```

```
    public void setL(int l) {
        if (l < 0) {
            throw new IllegalArgumentException("Signature length cannot be negative");
        }
        if (sigM != null && l > sigM.length) {
            throw new IllegalArgumentException("Signature length cannot exceed data length");
        }
        this.l = l;
```
